https://leetcode.com/problems/remove-duplicates-from-sorted-array/

class Solution {
public:
    int removeDuplicates(vector<int>& nums) {
        int i=0;
        int j=1;
        while(i<nums.size() && j<nums.size()){
            if(nums[i]==nums[j]){
                nums.erase(nums.begin()+j);
                continue;
            }i++;j++;
        } return  nums.size();
    }
};
